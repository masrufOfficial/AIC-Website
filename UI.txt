<script>
  document.getElementById("getLocation").addEventListener("click", function() {
    if (navigator.geolocation) {
      navigator.geolocation.getCurrentPosition(function(position) {
        const latitude = position.coords.latitude;
        const longitude = position.coords.longitude;
        
        // Fetch the address using reverse geocoding (e.g., using Google Maps API)
        // Set the location input field with the fetched address (you'll need an API to convert lat/long to address)
        document.getElementById("location").value = `Lat: ${latitude}, Long: ${longitude}`;
      });
    } else {
      alert("Geolocation is not supported by this browser.");
    }
  });
</script>
